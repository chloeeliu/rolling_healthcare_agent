```python
import pandas as pd

def cardiac_marker(stay_id):
    """
    Extract cardiac marker information for a specific ICU stay.
    
    This function retrieves troponin and NT-proBNP lab values for a patient
    during their ICU stay and provides clinical interpretation based on
    established thresholds.
    
    Parameters
    ----------
    stay_id : int
        The ICU stay identifier.
    
    Returns
    -------
    dict or None
        A dictionary containing the following keys:
        - 'stay_id': ICU stay identifier (int)
        - 'subject_id': patient identifier (int)
        - 'hadm_id': hospital admission identifier (int or None)
        - 'intime': ICU admission timestamp (datetime)
        - 'outtime': ICU discharge timestamp (datetime)
        - 'troponin_t_max': maximum Troponin T value during ICU stay (float or None)
        - 'troponin_t_unit': unit for Troponin T (str or None)
        - 'troponin_t_elevated': boolean indicating if Troponin T > 0.1 ng/mL
        - 'troponin_t_count': number of Troponin T measurements (int)
        - 'troponin_i_max': maximum Troponin I value during ICU stay (float or None)
        - 'troponin_i_unit': unit for Troponin I (str or None)
        - 'troponin_i_count': number of Troponin I measurements (int)
        - 'ntprobnp_max': maximum NT-proBNP value during ICU stay (float or None)
        - 'ntprobnp_unit': unit for NT-proBNP (str or None)
        - 'ntprobnp_elevated': boolean indicating if NT-proBNP > 1000 pg/mL
        - 'ntprobnp_count': number of NT-proBNP measurements (int)
        
        Returns None if no matching ICU stay is found.
    
    Notes
    -----
    Clinical thresholds used:
    - Troponin T elevated: > 0.1 ng/mL (indicative of myocardial injury)
    - NT-proBNP elevated: > 1000 pg/mL (suggestive of heart failure)
    
    Cardiac marker item IDs:
    - 50963: NTproBNP (pg/mL)
    - 51002: Troponin I (ng/mL)
    - 51003: Troponin T (ng/mL)
    - 52642: Troponin I (ng/mL)
    
    Examples
    --------
    >>> cardiac_marker(34081592)
    {'stay_id': 34081592, 'subject_id': 10067389, 'hadm_id': 23577021, 
     'troponin_t_max': 1.31, 'troponin_t_elevated': True, ...}
    """
    
    # Check if ICU stay exists
    stay_info = query_db(f"""
        SELECT stay_id, subject_id, hadm_id, intime, outtime 
        FROM mimiciv_icu.icustays 
        WHERE stay_id = {stay_id}
    """)
    
    if stay_info.empty:
        return None
    
    stay_info = stay_info.iloc[0]
    
    # Query for cardiac markers during the ICU stay
    sql = """
    SELECT i.stay_id, i.subject_id, i.hadm_id, i.intime, i.outtime,
           le.itemid, d.label, le.valuenum, le.valueuom, le.charttime
    FROM mimiciv_icu.icustays i
    JOIN mimiciv_hosp.labevents le ON i.hadm_id = le.hadm_id 
        AND le.charttime >= i.intime 
        AND le.charttime <= i.outtime
    JOIN mimiciv_hosp.d_labitems d ON le.itemid = d.itemid
    WHERE i.stay_id = {stay_id}
        AND le.itemid IN (50963, 51002, 51003, 52642)
        AND le.valuenum IS NOT NULL
    """.format(stay_id=stay_id)
    
    df = query_db(sql)
    
    result = {
        'stay_id': stay_id,
        'subject_id': int(stay_info['subject_id']),
        'hadm_id': int(stay_info['hadm_id']) if pd.notna(stay_info['hadm_id']) else None,
        'intime': stay_info['intime'],
        'outtime': stay_info['outtime'],
        'troponin_t_max': None,
        'troponin_t_unit': None,
        'troponin_t_elevated': False,
        'troponin_t_count': 0,
        'troponin_i_max': None,
        'troponin_i_unit': None,
        'troponin_i_count': 0,
        'ntprobnp_max': None,
        'ntprobnp_unit': None,
        'ntprobnp_elevated': False,
        'ntprobnp_count': 0
    }
    
    if not df.empty:
        # Troponin T (itemid 51003)
        troponin_t = df[df['itemid'] == 51003]
        if not troponin_t.empty:
            result['troponin_t_max'] = float(troponin_t['valuenum'].max())
            result['troponin_t_unit'] = troponin_t['valueuom'].iloc[0]
            result['troponin_t_count'] = len(troponin_t)
            result['troponin_t_elevated'] = result['troponin_t_max'] > 0.1
        
        # Troponin I (itemid 51002, 52642)
        troponin_i = df[df['itemid'].isin([51002, 52642])]
        if not troponin_i.empty:
            result['troponin_i_max'] = float(troponin_i['valuenum'].max())
            result['troponin_i_unit'] = troponin_i['valueuom'].iloc[0]
            result['troponin_i_count'] = len(troponin_i)
        
        # NT-proBNP (itemid 50963)
        ntprobnp = df[df['itemid'] == 50963]
        if not ntprobnp.empty:
            result['ntprobnp_max'] = float(ntprobnp['valuenum'].max())
            result['ntprobnp_unit'] = ntprobnp['valueuom'].iloc[0]
            result['ntprobnp_count'] = len(ntprobnp)
            result['ntprobnp_elevated'] = result['ntprobnp_max'] > 1000
    
    return result
```

```python
import pandas as pd
import numpy as np

def cardiac_marker(stay_id):
    """
    Extract cardiac marker information for a specific ICU stay.
    
    This function retrieves troponin and NT-proBNP/BNP lab values for a patient
    during their ICU stay from both labevents (hospital) and chartevents (ICU)
    sources, and returns the maximum value if any marker is elevated.
    
    Parameters
    ----------
    stay_id : int
        The ICU stay identifier.
    
    Returns
    -------
    float or np.nan
        The maximum cardiac marker value if any marker is elevated, otherwise np.nan.
        Returns np.nan if no matching ICU stay is found.
    
    Notes
    -----
    Clinical thresholds used:
    - Troponin T elevated: > 0.1 ng/mL (indicative of myocardial injury)
    - NT-proBNP/BNP elevated: > 1000 pg/mL (suggestive of heart failure)
    
    Cardiac marker item IDs:
    labevents: 50963 (NTproBNP), 51002 (Troponin I), 51003 (Troponin T), 52642 (Troponin I)
    chartevents: 227429 (Troponin-T), 227446 (BNP), 225622 (BNP)
    """
    
    # Check if ICU stay exists
    stay_info = query_db(f"""
        SELECT stay_id, subject_id, hadm_id, intime, outtime 
        FROM mimiciv_icu.icustays 
        WHERE stay_id = {stay_id}
    """)
    
    if stay_info.empty:
        return np.nan
    
    stay_info = stay_info.iloc[0]
    
    # Query for cardiac markers from labevents during the ICU stay
    sql_lab = """
    SELECT le.itemid, le.valuenum
    FROM mimiciv_hosp.labevents le
    JOIN mimiciv_icu.icustays i ON le.hadm_id = i.hadm_id
    WHERE i.stay_id = {stay_id}
        AND le.charttime >= i.intime 
        AND le.charttime <= i.outtime
        AND le.itemid IN (50963, 51002, 51003, 52642)
        AND le.valuenum IS NOT NULL
    """.format(stay_id=stay_id)
    
    df_lab = query_db(sql_lab)
    
    # Query for cardiac markers from chartevents during the ICU stay
    sql_chart = """
    SELECT c.itemid, c.valuenum
    FROM mimiciv_icu.chartevents c
    JOIN mimiciv_icu.icustays i ON c.stay_id = i.stay_id
    WHERE i.stay_id = {stay_id}
        AND c.charttime >= i.intime 
        AND c.charttime <= i.outtime
        AND c.itemid IN (227429, 227446, 225622)
        AND c.valuenum IS NOT NULL
    """.format(stay_id=stay_id)
    
    df_chart = query_db(sql_chart)
    
    # Combine both sources
    df = pd.concat([df_lab, df_chart], ignore_index=True)
    
    if df.empty:
        return np.nan
    
    # Troponin T (51003, 227429) elevated if > 0.1
    troponin_t = df[df['itemid'].isin([51003, 227429])]
    troponin_t_elevated = not troponin_t.empty and troponin_t['valuenum'].max() > 0.1
    
    # NT-proBNP/BNP (50963, 227446, 225622) elevated if > 1000
    ntprobnp = df[df['itemid'].isin([50963, 227446, 225622])]
    ntprobnp_elevated = not ntprobnp.empty and ntprobnp['valuenum'].max() > 1000
    
    if troponin_t_elevated or ntprobnp_elevated:
        return float(df['valuenum'].max())
    else:
        return np.nan
```

### Key Implementation Notes
- Function queries BOTH labevents AND chartevents for cardiac markers
- Returns single numeric value: max cardiac marker value if elevated, otherwise np.nan
- Returns np.nan for invalid stays or when no elevated markers exist
- Troponin T uses itemids: 51003 (labevents), 227429 (chartevents)
- Troponin I uses itemids: 51002, 52642 (labevents only)
- NT-proBNP/BNP uses itemids: 50963 (labevents), 227446, 225622 (chartevents)
- Only includes values within ICU stay time range (charttime >= intime AND charttime <= outtime)
- Some stays have markers only in chartevents (e.g., 31392237, 37098798)
- Some stays have markers in both sources (e.g., 34081592, 35762622)
